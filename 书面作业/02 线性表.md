void deleteNode(ListNode* p,int val){
    if(!p) return ;
    ListNode* cur=p;
    ListNode* st=p;
    bool flag=true;
    while(cur&&cur!=st){
        flag=false;
        if(cur->val==val){
            if(cur->next==cur&&cur->pre==cur){
                delete cur;
                return nullptr;
            }
            cur->pre->next=cur->next;
            cur->next->pre=cur->pre;
            if(cur==p){
                p=cur->next;
                st=p;
            }
            delete cur;
        }
        cur=cur->next;
    }
    return p;
}
